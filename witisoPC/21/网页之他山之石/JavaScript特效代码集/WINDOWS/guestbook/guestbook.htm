以下代码加入HTML的<body>区,注释行可以去掉：:

<!--代码由此开始，检测填写项目的完整，不填写或填写默认值都会警告。-->
<SCRIPT LANGUAGE = "JavaScript">

function formCheck()
{
if (document.theform.email.value.indexOf("@") == -1 ||
document.theform.email.value == "")
{
alert("请填写完整的E-MAIL!");
return false;
}

if (document.theform.City.value == "")
{
alert("请填写完整表单!");
return false;
}
if (document.theform.City.value == "所在城市:")
{
alert("请填写所在城市!");
return false;
}
if (document.theform.NICK_name.value == "网上昵称:")
{
alert("请填写你的网上昵称!");
return false;
}
if (document.theform.NICK_name.value == "")
{
alert("请填写你的网上昵称!");
return false;
}
if (document.theform.HOME_Page.value == "http://")
{
alert("请填写你的网址!");
return false;
}
if (document.theform.HOME_Page.value == "")
{
alert("请填写你的网址!");
return false;
}
if (document.theform.Content.value == "Content:")
{
alert("请填写留言内容!");
return false;
}
if (document.theform.Content.value == "")
{
alert("请填写留言内容!");
return false;
}
}
</script>
