独有特色的倒计时脚本
一个倒计时脚本，时间到了后会显示欢迎词，超过倒计时日期后，信息就不再显示了。

（如果本行没有图片显示说明：时间在２０００年１２月２５日以后）
本例显示距离２０００年圣诞节的倒计时时间。

 

如下的脚本加入ＨＴＭＬ的<body>区：

<script>
//以下参数第一行是显示的特殊日名称，后面要引用的。
//参数第二行是特殊日到了后显示的欢迎词。
//特殊日过了后就不显示了。


var before="圣诞节"
var current="圣诞节到了，祝大家圣诞快乐！"
var montharray=new Array("Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dec")

function countdown(yr,m,d){
var today=new Date()
var todayy=today.getYear()
var todaym=today.getMonth()
var todayd=today.getDate()
var todaystring=montharray[todaym]+" "+todayd+", "+todayy
var futurestring=montharray[m-1]+" "+d+", "+yr
var difference=(Math.round((Date.parse(futurestring)-Date.parse(todaystring))/(24*60*60*1000))*1)
var dstring=difference.toString()
if (difference==0)
document.write(current)
else if (difference>0){
document.write("距离"+before)
for (i=0;i<=dstring.length-1;i++)
document.write('<img src="c'+dstring.charAt(i)+'.gif">')
document.write('天')
}
}
//倒计时的日期在这里。
countdown(2000,12,25)
</script>

注释：自己用任意的图片（c0.gif--c9.gif代替本例的图片即可。）
