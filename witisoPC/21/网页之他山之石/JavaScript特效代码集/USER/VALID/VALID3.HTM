表单递交HTML语句屏蔽脚本

代码加入HTML的<BODY>区：

<script language="javascript">
function htmlOut(){
txt=document.myform.html.value;
ctr=0;
while ((txt.indexOf("<")>-1)&&(ctr<4)){
pos=txt.indexOf("<");
txt=txt.substring(0,pos)+"&lt;"+txt.substring(pos+1,txt.length);
ctr++;
}
while (txt.indexOf(">")>-1){
pos=txt.indexOf(">");
txt=txt.substring(0,pos)+"&gt;"+txt.substring(pos+1,txt.length);
}
document.myform.html.value=txt;
}
</script>
<form name="myform" method="post" enctype="text/plain" onsubmit="htmlOut()" action="mailto:mail@mail.com">
<DIV ALIGN=CENTER>

<textarea name="html" cols="60" rows="15"></textarea>
<BR>
<INPUT TYPE="submit" NAME="submit" VALUE=" Submit " >
</DIV>
</form>
