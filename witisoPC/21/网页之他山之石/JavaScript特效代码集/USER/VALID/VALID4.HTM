表单递交内容非空检测脚本


如下代码加入HTML的<BODY>区即可：

<script language="javascript">
function noEntry(){
mt=document.myform.content.value;
if (mt.length<1){
alert("递交必须要有内容。");
return false;
}
}
</script>


<form name="myform" method="post" enctype="text/plain" onsubmit="return noEntry()" action="mailto:mail@mail.com?subject=递交表单">
<DIV ALIGN=CENTER>
<textarea name="content" cols="60" rows="15"></textarea>
<BR>
<INPUT TYPE="submit" NAME="submit" VALUE=" Submit " >
</DIV>
</form>
