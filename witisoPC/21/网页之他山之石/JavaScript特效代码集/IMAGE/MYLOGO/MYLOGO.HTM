MyLogo 固定的LOGO脚本

<html>
<head>
<title></title>
</head>
<style>
.logo
{
position:absolute;
}
</style>
</head>

<script Language="Javascript">

xySpace = 200;

function start()
{

if (navigator.appName == "Netscape")
{
style_logo = "document.logo";
yOffset = "self.innerHeight + self.pageYOffset - xySpace ";
xOffset = "self.innerWidth - xySpace ";
setMAJ();
}
else
{
style_logo = "document.all.logo.style";
onscroll = MAJLogo;
yOffset = "document.body.offsetHeight + document.body.scrollTop - xySpace ";
xOffset = "document.body.offsetWidth - xySpace ";
MAJLogo();
}
}
function MAJLogo()
{
eval(style_logo + ".top = " + yOffset);
eval(style_logo + ".left = " + xOffset);
}

function setMAJ(evt)
{
setInterval('MAJLogo()',10);
}
</script>


<body bgcolor="#FFFFFF" onLoad="start()">
