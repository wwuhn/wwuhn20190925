连接说明浮动条脚本

［论坛］

［关于］


将鼠标放在连接上看看，所有连接都通向首页。


将如下代码加入HTML的<BODY>区：

<Script Language="Javascript">
<!-- Hiding
bname=navigator.appName;
bversion=parseInt(navigator.appVersion)
if (bname=="Netscape")
brows=true
else
brows=false
var x=0;
var link=new Array();

function dspl(msg,bgcolor,dtop,delft){
this.msg=msg;
this.bgcolor=bgcolor;
this.dtop=dtop;
this.dleft=delft;
}


link[0]=new dspl('［宗旨］<BR>内容','bisque',150,430)

link[1]=new dspl('［论坛］<BR>内容','bisque',175,430)

link[2]=new dspl('［栏目］<BR>内容','bisque',200,430)

link[3]=new dspl('［关于］<BR>内容','bisque',225,430)

//兰色部分是浮动条的背景色，红色部分是浮动条的绝对位置，前面是Y轴，后面的是X轴。根据连接的具体位置设置这个位置，如果你用DW2，可以打开Ruler（坐标线），坐标位置非常容易找到。

// 不要改动下面的代码

function don(x){
if ((bname=="Netscape" && bversion>=4) || (bname=="Microsoft Internet Explorer" && bversion>=4)){
if (brows){
with(link[x]){
document.layers['linkex'].bgColor=bgcolor;
document.layers['linkex'].document.writeln(msg);
document.layers['linkex'].document.close();
document.layers['linkex'].top=dtop;
document.layers['linkex'].left=dleft;
}
document.layers['linkex'].visibility="show";
}
else{
with(link[x]){
linkex.innerHTML=msg;
linkex.style.top=dtop;
linkex.style.left=dleft;
linkex.style.background=bgcolor;
}
linkex.style.visibility="visible";
}
}
}

function doff(){
if ((bname=="Netscape" && bversion>=4) || (bname=="Microsoft Internet Explorer" && bversion>=4)){
if (brows)
document.layers['linkex'].visibility="hide";
else
linkex.style.visibility="hidden";
}
}

// done hiding -->
</Script>

//以下红色表示浮动条的宽度，IE和NS分别用<div>标签和<layer>标签。

<Div id="linkex" style="position: absolute; visibility: hidden; width=19%">
</Div>
<Layer name="linkex" visibility="hide" width=19%>
</Layer>

//下面是每个连接的URL及打开的窗口方式(_top,_blank等)

<a href="index1.html" onmouseover="don(0)" onmouseout="doff()" target="_top >［宗旨］</a>
<BR>
<a href="index2.html" onmouseover="don(1)" onmouseout="doff()" target="_top >［论坛］</a>
<br>
<a href="index3.html" onmouseover="don(2)" onmouseout="doff()" target="_top >［栏目］</a>
<BR>
<a href="index4.html" onmouseover="don(3)" onmouseout="doff()" target="_top">［关于］</a>
